013package org.mmtk.utility.options;
014
015import static org.mmtk.policy.immix.ImmixConstants.DEFAULT_SIMPLE_SPILL_THRESHOLD;
016
017public class DefragSimpleSpillThreshold extends org.vmutil.options.FloatOption {
018  /**
019   * Create the option.
020   */
021  public DefragSimpleSpillThreshold() {
022    super(Options.set, "Defrag Simple Spill Threshold",
023          "Blocks with this fraction spilled will be defrag sources",
024          DEFAULT_SIMPLE_SPILL_THRESHOLD);
025  }
026
027  /**
028   * Ensure the value is valid.
029   */
030  @Override
031  protected void validate() {
032    failIf((this.value <= 0 || this.value > 1.0), "Ratio must be a float between 0 and 1");
033  }
034}
