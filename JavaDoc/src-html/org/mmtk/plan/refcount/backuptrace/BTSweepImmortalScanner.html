013package org.mmtk.plan.refcount.backuptrace;
014
015import org.mmtk.plan.refcount.RCHeader;
016import org.mmtk.utility.alloc.LinearScan;
017
018import org.vmmagic.pragma.*;
019import org.vmmagic.unboxed.*;
020
021/**
022 * This class is used in backup tracing to sweep and zero any dead 'immortal' objects.
023 */
024@Uninterruptible
025public final class BTSweepImmortalScanner extends LinearScan {
026
027  /**
028   * Scan an object during sweep.
029   *
030   * @param object The source of the reference.
031   */
032  @Override
033  @Inline
034  public void scan(ObjectReference object) {
035    if (!RCHeader.isMarked(object)) {
036      // do nothing
037    } else {
038      RCHeader.clearMarked(object);
039    }
040  }
041}
