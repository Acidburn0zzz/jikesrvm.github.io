013package org.mmtk.plan.refcount;
014
015import static org.mmtk.policy.SegregatedFreeListSpace.MAX_FREELIST_OBJECT_BYTES;
016import org.mmtk.plan.StopTheWorldConstraints;
017
018import org.vmmagic.pragma.*;
019
020/**
021 * This class and its subclasses communicate to the host VM/Runtime
022 * any features of the selected plan that it needs to know.  This is
023 * separate from the main Plan/PlanLocal class in order to bypass any
024 * issues with ordering of static initialization.
025 */
026@Uninterruptible
027public class RCBaseConstraints extends StopTheWorldConstraints {
028  @Override
029  public int gcHeaderBits() { return RCHeader.GLOBAL_GC_BITS_REQUIRED; }
030  @Override
031  public int gcHeaderWords() { return RCHeader.GC_HEADER_WORDS_REQUIRED; }
032  @Override
033  public boolean needsObjectReferenceWriteBarrier() { return true; }
034  @Override
035  public int maxNonLOSDefaultAllocBytes() { return MAX_FREELIST_OBJECT_BYTES; }
036  @Override
037  public boolean objectReferenceBulkCopySupported() { return true; }
038
039  /** @return {@code true} if we are building for generational reference counting */
040  public boolean buildForGenRC() { return false; }
041}
