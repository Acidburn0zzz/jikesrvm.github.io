013package gnu.classpath;
014
015import org.vmmagic.unboxed.Address;
016
017import org.jikesrvm.VM;
018
019/**
020 * Library support interface of Jikes RVM
021 */
022public class JikesRVMSupport {
023  public static Address getAddressFromPointer(Pointer pointer) {
024    if (VM.BuildFor32Addr)
025      return Address.fromIntSignExtend(((Pointer32)pointer).data);
026    else
027      return Address.fromLong(((Pointer64)pointer).data);
028  }
029
030  public static Pointer getPointerFromAddress(Address address) {
031    if (VM.BuildFor32Addr)
032      return new Pointer32(address.toInt());
033    else
034      return new Pointer64(address.toLong());
035  }
036}
